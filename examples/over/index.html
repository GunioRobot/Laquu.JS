Code

jQuery(function(){
    laquu("#over2 tr").over({
        hoverClass: "on",
        onHover: function() {
            laquu("th", this).text("Hello");
            laquu("td", this).text("World!");
        },
        onOut: function() {
            laquu("th", this).text("Table head");
            laquu("td", this).text("Table data");
        }
    });
});
